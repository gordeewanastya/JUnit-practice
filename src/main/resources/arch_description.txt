interface IReader
    List<String> read()

class CSVReader implements Reader
    List<String> read() -> returns list of entities which was read from the csv file

interface IMapper<T>
    <T> mapFromString(List<String> value)

class EmployeeMapper implements Mapper<Employee>
    Employee mapFromString(List<String> value)

interface IValidator<T>
    boolean isValid(<T> entity)

class Validator implements IValidator<Employee>
    isValid(Employee employee)

interface IEmployeeRepo
    saveEmployee(Employee employee)

class EmployeeRepo implement IEmployeeRepo
    saveEmployee(Employee employee) -> we'll use EmployeeValidator here
